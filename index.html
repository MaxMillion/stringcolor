String to Color
jQuery plugin

any string
any string

jQuery Plugin:

/**
 * Generate hex color code from a string.
 *
 * @param {String} str
 * @return {String}
 */
$.string_to_color(string);

/**
 * Set one or more CSS properties for the set
 * of matched elements.
 *
 * @param {String|Array} property
 * @param {String} str
 */
$(selector).string_to_color(property, string);

Example:

// Generate hex color.
var c = $.string_to_color("foo");
alert(c); // #0073ad

// Change border color and font color.
$(".foo").string_to_color(
	["border-color", "color"],
	"hello" // #ff00b9
);
